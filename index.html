	class DBQuery extends AsyncResource {
		constructor(db) {
			super('DBQuery');
			this.db = db;
		}

		getInfo(query, callback) {
			this.db.get(query, (err, data) => {
				this.runInAsyncScope(callback, null, err, data);
			});
		}

		close() {
			this.db = null;
			this.emitDestroy();
		}
	}
